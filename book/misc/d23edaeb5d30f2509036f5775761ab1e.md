    <script>

        var salt = "ThisLooksInteresting";
        const hash = "d184f2921b373367fe11cf4529291f9e4e43e50eb8bb9df134654abbf5c1172f";

        async function hashPassword(password) {
            const encoder = new TextEncoder();
            const data = encoder.encode(salt + password);
            const hashBuffer = await crypto.subtle.digest("SHA-256", data);
            const hashArray = Array.from(new Uint8Array(hashBuffer));
            const hashHex = hashArray.map(byte => byte.toString(16).padStart(2, "0")).join("");
            console.log(hashHex);
            return hashHex;
        }

        async function checkPassword() {
            const input = document.getElementById("passwordInput").value;
            const hashedInput = await hashPassword(input);

            if (hashedInput === hash) {
                document.getElementById("message").textContent = "You got it! Let Alex know what your answer was for your reward.";
            } else {
                document.getElementById("message").textContent = "You really should not press red buttons.";
            }
        }
    </script>
